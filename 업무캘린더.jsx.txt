import React, { useState, useEffect, useMemo } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth';
import { getFirestore, doc, setDoc, onSnapshot } from 'firebase/firestore';
import { ChevronLeft, ChevronRight, Calendar, AlertTriangle, Search, CheckCircle, BookOpen, RotateCcw, CloudSun, Trash2, Edit2, Check, Clock } from 'lucide-react';

const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

export default function App() {
  const [currentMonth, setCurrentMonth] = useState(new Date());
  const [selectedDate, setSelectedDate] = useState(new Date());
  const [user, setUser] = useState(null);
  const [data, setData] = useState({ todos: {}, instructions: {}, fixedTasksDone: {}, inspectionSchedules: ["7/10 현장 정기 점검", "7/15 장비 안전검사", "7/22 타설 현장 합동 점검"] });
  const [searchTerm, setSearchTerm] = useState('');
  
  const [editingInstId, setEditingInstId] = useState(null);
  const [editingInstText, setEditingInstText] = useState('');
  const [editingTodoId, setEditingTodoId] = useState(null);
  const [editingTodoText, setEditingTodoText] = useState('');
  const [isEditingInspection, setIsEditingInspection] = useState(false);
  const [editInspectionText, setEditInspectionText] = useState('');

  const formatDate = (date) => {
    const y = date.getFullYear();
    const m = String(date.getMonth() + 1).padStart(2, '0');
    const d = String(date.getDate()).padStart(2, '0');
    return `${y}-${m}-${d}`;
  };

  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (u) => setUser(u));
    return () => unsubscribe();
  }, []);

  useEffect(() => {
    if (!user) return;
    const docRef = doc(db, 'artifacts', appId, 'users', user.uid, 'appData', 'state');
    const unsubscribe = onSnapshot(docRef, (docSnap) => {
      if (docSnap.exists()) {
        const newData = docSnap.data();
        
        // 점검 일정 자동 이동 로직
        const today = new Date();
        today.setHours(0, 0, 0, 0);
        let hasChanges = false;
        let newSchedules = [...(newData.inspectionSchedules || [])];
        let newTodos = { ...newData.todos };

        newData.inspectionSchedules?.forEach((s) => {
          const match = s.match(/^(\d+)\/(\d+)\s*(.*)/);
          if (match) {
            const month = parseInt(match[1]) - 1;
            const day = parseInt(match[2]);
            const targetDate = new Date(today.getFullYear(), month, day);
            if (targetDate <= today) {
              const key = formatDate(targetDate);
              const task = { id: Date.now() + Math.random(), text: match[3], done: false };
              newTodos[key] = [...(newTodos[key] || []), task];
              newSchedules = newSchedules.filter(item => item !== s);
              hasChanges = true;
            }
          }
        });

        if (hasChanges) {
          const updatedData = { ...newData, inspectionSchedules: newSchedules, todos: newTodos };
          saveData(updatedData);
        } else {
          setData(newData);
        }
      }
    }, (err) => console.error("Firestore sync error:", err));
    return () => unsubscribe();
  }, [user]);

  const saveData = async (newData) => {
    if (!user) return;
    const docRef = doc(db, 'artifacts', appId, 'users', user.uid, 'appData', 'state');
    await setDoc(docRef, newData, { merge: true });
    setData(newData);
  };

  const { todos, instructions, fixedTasksDone, inspectionSchedules } = data;
  const dateKey = formatDate(selectedDate);

  const getRecurringTasks = (date) => {
    const day = date.getDay();
    const dateNum = date.getDate();
    if (day === 0) return [];
    
    const tasks = [];
    if (day >= 1 && day <= 6) tasks.push({id: "band-tbm", text: "밴드 TBM 업로드"});
    if (day >= 1 && day <= 6 && date <= new Date(2026, 8, 15)) tasks.push({id: "health-check", text: "체감온도기록지/혈압지"});
    if (day === 4) tasks.push({id: "holiday-plan", text: "휴일작업계획서/위험성평가 제출"});
    if (day === 1) tasks.push({id: "holiday-report", text: "휴일작업 작업후 보고서"});
    if (dateNum >= 10 && dateNum <= 12 && day >= 1 && day <= 5) tasks.push({id: "risk-eval-2", text: "위험성평가(2주차)"});
    if (dateNum >= 23 && dateNum <= 25) tasks.push({id: "next-month-risk-1", text: "다음달 1회차 위험성평가 작성 및 제출"});
    if (dateNum >= 23 && dateNum <= 28) tasks.push({id: "safety-edu", text: "📢 정기안전교육"});
    return tasks;
  };

  const pendingTasks = useMemo(() => {
    const list = [];
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    const startOfMonth = new Date(today.getFullYear(), today.getMonth(), 1);

    for (let d = new Date(startOfMonth); d <= today; d.setDate(d.getDate() + 1)) {
      const key = formatDate(d);
      getRecurringTasks(d).forEach(task => { 
        if (!fixedTasksDone[`${key}-${task.id}`]) {
          list.push({ date: key, text: task.text, id: `${key}-${task.id}`, isFixed: true }); 
        }
      });
      (todos[key] || []).forEach(t => { 
        if (!t.done) list.push({ date: key, text: t.text, id: t.id, isFixed: false }); 
      });
    }
    return list.filter(t => t.text.toLowerCase().includes(searchTerm.toLowerCase()));
  }, [data, searchTerm, fixedTasksDone, todos]);

  return (
    <div className="min-h-screen bg-slate-300 p-4 md:p-6 text-slate-800 font-sans">
      <style>{`
        @keyframes marquee { 0% { transform: translateX(100%); } 100% { transform: translateX(-100%); } }
        .animate-marquee { display: inline-block; animation: marquee 25s linear infinite; }
      `}</style>
      <div className="max-w-7xl mx-auto space-y-4">
        <div className="flex items-center gap-2 bg-white p-2 rounded-xl border border-slate-400 overflow-hidden shadow-sm">
          <div className="flex-1 overflow-hidden whitespace-nowrap">
            <div className="animate-marquee inline-block text-xs md:text-sm text-blue-700 font-bold">
                {(inspectionSchedules || []).map((s, i) => <span key={i} className="mx-4">◈ {s}</span>)}
            </div>
          </div>
          <button onClick={() => { setIsEditingInspection(true); setEditInspectionText((inspectionSchedules || []).join('\n')); }} className="p-1 hover:bg-slate-100 rounded text-slate-500"><Edit2 size={16}/></button>
        </div>

        {isEditingInspection && (
          <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-4">
            <div className="bg-white p-6 rounded-2xl w-full max-w-md shadow-2xl">
              <h3 className="font-bold mb-4">점검 일정 수정 (M/D 내용 형식)</h3>
              <textarea value={editInspectionText} onChange={(e) => setEditInspectionText(e.target.value)} className="w-full h-40 p-2 border rounded-lg mb-4 text-sm" placeholder="예: 7/10 현장 정기 점검" />
              <div className="flex gap-2 justify-end">
                <button onClick={() => setIsEditingInspection(false)} className="px-4 py-2 bg-slate-200 rounded-lg text-sm">취소</button>
                <button onClick={() => { saveData({...data, inspectionSchedules: editInspectionText.split('\n').filter(s => s.trim())}); setIsEditingInspection(false); }} className="px-4 py-2 bg-blue-600 text-white rounded-lg text-sm">저장</button>
              </div>
            </div>
          </div>
        )}

        <header className="flex flex-col md:flex-row items-center justify-between bg-white p-4 rounded-2xl shadow-sm border border-slate-400 gap-4">
          <h1 className="text-xl md:text-2xl font-bold flex items-center gap-3 text-slate-800">
            <Calendar className="text-blue-600"/> 업무캘린더
            <a href="https://www.weather.go.kr" target="_blank" rel="noreferrer"><CloudSun className="text-orange-500" size={24}/></a>
          </h1>
          <div className="flex items-center gap-2">
            <button onClick={() => { const t = new Date(); setCurrentMonth(t); setSelectedDate(t); }} className="px-3 py-1.5 bg-blue-600 text-white rounded-lg text-sm font-bold flex items-center gap-1 hover:bg-blue-700 border border-blue-700 shadow-sm"><RotateCcw size={14}/> 오늘</button>
            <div className="relative"><Search className="absolute left-2 top-2 text-slate-500" size={16}/><input className="bg-slate-100 border border-slate-400 rounded-lg py-1.5 pl-8 pr-4 text-sm w-36 md:w-48" placeholder="업무 검색..." onChange={(e) => setSearchTerm(e.target.value)} /></div>
            <button onClick={() => setCurrentMonth(new Date(currentMonth.setMonth(currentMonth.getMonth() - 1)))} className="p-2 bg-slate-200 hover:bg-slate-300 rounded-lg border border-slate-400"><ChevronLeft size={18}/></button>
            <span className="font-bold text-sm w-24 text-center">{currentMonth.getFullYear()}년 {currentMonth.getMonth() + 1}월</span>
            <button onClick={() => setCurrentMonth(new Date(currentMonth.setMonth(currentMonth.getMonth() + 1)))} className="p-2 bg-slate-200 hover:bg-slate-300 rounded-lg border border-slate-400"><ChevronRight size={18}/></button>
          </div>
        </header>

        <div className="grid grid-cols-1 lg:grid-cols-4 gap-4">
          <div className="lg:col-span-3 bg-white p-4 rounded-2xl shadow-sm border border-slate-400 h-[650px] flex flex-col overflow-hidden">
             <div className="grid grid-cols-7 gap-1 text-center text-[10px] md:text-xs font-bold text-slate-500 mb-2">
              {['일','월','화','수','목','금','토'].map((d, i) => <div key={d} className={i===0?'text-red-600':i===6?'text-blue-700':''}>{d}</div>)}
            </div>
            <div className="grid grid-cols-7 grid-rows-6 gap-1 flex-1">
              {[...Array(new Date(currentMonth.getFullYear(), currentMonth.getMonth(), 1).getDay()).keys()].map(i => <div key={`e-${i}`}/>)}
              {[...Array(new Date(currentMonth.getFullYear(), currentMonth.getMonth() + 1, 0).getDate()).keys()].map(d => {
                const date = new Date(currentMonth.getFullYear(), currentMonth.getMonth(), d + 1);
                const day = date.getDay();
                const key = formatDate(date);
                return (
                  <button key={d} onClick={() => setSelectedDate(date)} 
                    className={`rounded-lg border p-1 flex flex-col items-start h-20 md:h-24 overflow-hidden ${date.toDateString() === selectedDate.toDateString() ? 'bg-blue-50 border-blue-400' : 'bg-slate-50 border-slate-300'} transition`}>
                    <span className={`text-[10px] font-bold ${day === 0 ? 'text-red-600' : day === 6 ? 'text-blue-700' : 'text-slate-800'}`}>{d + 1}</span>
                    <div className="flex-1 w-full overflow-hidden space-y-0.5">
                      {(instructions[key] || []).map((t, i) => <div key={`inst-${i}`} className="text-[8px] bg-violet-200 text-violet-900 rounded px-0.5 w-full truncate">★ {t.text}</div>)}
                      {(todos[key] || []).map((t, i) => <div key={`todo-${i}`} className="text-[8px] bg-emerald-200 text-emerald-900 rounded px-0.5 w-full truncate">● {t.text}</div>)}
                    </div>
                  </button>
                );
              })}
            </div>
          </div>

          <div className="lg:col-span-1 bg-white p-6 rounded-2xl shadow-sm border border-slate-400 space-y-4 h-[650px] overflow-y-auto">
            <h3 className="font-bold text-sm border-b border-slate-300 pb-2 text-slate-800">{selectedDate.getMonth() + 1}월 {selectedDate.getDate()}일</h3>
            
            <section className="bg-red-50 p-3 rounded-lg border border-red-300">
              <h4 className="text-[11px] font-bold text-red-800 flex items-center gap-1 mb-2"><AlertTriangle size={12}/> 미완료 대기소</h4>
              <div className="max-h-24 overflow-y-auto space-y-1">
                 {pendingTasks.map((p, i) => <div key={i} className="text-[10px] text-red-900 truncate">· [{p.date.slice(5)}] {p.text}</div>)}
              </div>
            </section>

            <section className="border-t border-slate-300 pt-3">
              <h4 className="text-[11px] font-bold text-emerald-800 flex items-center gap-1 mb-2"><CheckCircle size={12}/> 오늘 해야할 일</h4>
              <input className="w-full bg-white border-2 border-emerald-400 rounded-lg p-4 text-sm focus:border-emerald-600 focus:outline-none mb-4" placeholder="오늘의 할 일 추가..." onKeyDown={(e) => { if(e.key==='Enter' && e.target.value) { saveData({...data, todos: {...todos, [dateKey]: [...(todos[dateKey]||[]), {id: Date.now(), text: e.target.value, done: false}]}}); e.target.value='';}}} />
              <div className="space-y-2">
                {getRecurringTasks(selectedDate).map((task) => {
                  const done = !!fixedTasksDone[`${dateKey}-${task.id}`];
                  return (
                    <div key={`fixed-${task.id}`} className="flex items-center gap-2 text-xs">
                      <input type="checkbox" checked={done} onChange={() => saveData({...data, fixedTasksDone: {...fixedTasksDone, [`${dateKey}-${task.id}`]: !done}})} />
                      <span className={done ? 'line-through text-slate-500' : ''}>{task.text}</span>
                    </div>
                  );
                })}
                {(todos[dateKey] || []).map(t => (
                  <div key={t.id} className="flex items-center gap-2 text-xs">
                    <input type="checkbox" checked={t.done} onChange={() => saveData({...data, todos: {...todos, [dateKey]: todos[dateKey].map(i => i.id === t.id ? {...i, done: !t.done} : i)}})} />
                    {editingTodoId === t.id ? (
                      <div className="flex-1 flex gap-1"><input value={editingTodoText} onChange={e => setEditingTodoText(e.target.value)} className="border rounded p-1 flex-1"/> <button onClick={() => { saveData({...data, todos: {...todos, [dateKey]: todos[dateKey].map(i => i.id === t.id ? {...i, text: editingTodoText} : i)}}); setEditingTodoId(null);}}><Check size={14}/></button></div>
                    ) : (
                      <>
                        <span className={t.done ? 'line-through text-slate-500 flex-1' : 'flex-1'}>{t.text}</span>
                        <button onClick={() => {setEditingTodoId(t.id); setEditingTodoText(t.text);}}><Edit2 size={12}/></button>
                        <button onClick={() => saveData({...data, todos: {...todos, [dateKey]: todos[dateKey].filter(i => i.id !== t.id)}})}><Trash2 size={12}/></button>
                      </>
                    )}
                  </div>
                ))}
              </div>
            </section>

            <section className="border-t border-slate-300 pt-3">
              <h4 className="text-[11px] font-bold text-violet-800 flex items-center gap-1 mb-2"><BookOpen size={12}/> 소장님 지시사항</h4>
              <input className="w-full bg-white border-2 border-violet-400 rounded-lg p-4 text-sm focus:border-violet-600 focus:outline-none mb-4" placeholder="지시사항 입력..." onKeyDown={(e) => { if(e.key==='Enter' && e.target.value) { saveData({...data, instructions: {...instructions, [dateKey]: [...(instructions[dateKey]||[]), {id: Date.now(), text: e.target.value}]}}); e.target.value='';}}} />
              <div className="mt-2 space-y-1">
                {(instructions[dateKey] || []).map(t => (
                  <div key={t.id} className="flex items-center justify-between text-[10px] text-violet-900 bg-violet-100 p-2 rounded">
                    {editingInstId === t.id ? (
                      <div className="flex-1 flex gap-1"><input value={editingInstText} onChange={e => setEditingInstText(e.target.value)} className="border rounded p-1 flex-1"/><button onClick={() => { saveData({...data, instructions: {...instructions, [dateKey]: instructions[dateKey].map(i => i.id === t.id ? {...i, text: editingInstText} : i)}}); setEditingInstId(null);}}><Check size={14}/></button></div>
                    ) : (
                      <>
                        <span>★ {t.text}</span>
                        <div className="flex gap-1">
                          <button onClick={() => {setEditingInstId(t.id); setEditingInstText(t.text);}}><Edit2 size={12}/></button>
                          <button onClick={() => saveData({...data, instructions: {...instructions, [dateKey]: instructions[dateKey].filter(i => i.id !== t.id)}})}><Trash2 size={12}/></button>
                        </div>
                      </>
                    )}
                  </div>
                ))}
              </div>
            </section>
          </div>
        </div>
      </div>
    </div>
  );
}
